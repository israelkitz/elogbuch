Autenticação — Cloudflare Worker

Cole aqui a URL do seu Worker para ativar a autenticação segura. Sem isso, qualquer pessoa que saiba da URL pode tentar acessar.

Como funciona: a senha nunca fica no código HTML. Quando alguém tenta entrar, o app envia a senha digitada para o seu Worker no Cloudflare, que compara com a senha guardada lá e responde só "certo" ou "errado". Nem você consegue ver a senha olhando o código-fonte.

'Pré-laudo liberado' : c.statusLaudo === 'revisado' ? 'Revisado pelo preceptor' : ''; sheet.appendRow([c.id,c.nome,c.anoRes, c.dataExame,c.dataLaudo,c.local,c.paciente, c.regpaciente||'',c.dtnasc||'',c.idade||'', c.modal,c.area,c.nivel,c.tipo,c.dif,sl, c.preceptor,c.diag,c.nota, c.prelaudoTxt||'',c.laudoRevTxt||'']); added++; }); return ContentService.createTextOutput( callback+'('+JSON.stringify({ok:true,count:added})+')' ).setMimeType(ContentService.MimeType.JAVASCRIPT); } catch(err) { return ContentService.createTextOutput( callback+'('+JSON.stringify({ok:false,error:err.message})+')' ).setMimeType(ContentService.MimeType.JAVASCRIPT); } }
